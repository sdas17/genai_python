 The Professional Way to Set Up Python

 # Step 1: Check your Python version (must be 3.10+)
python --version        # or python3 --version

python -m venv venv  

# Step 3: Activate it
source venv/bin/activate     # Mac/Linux
venv\Scripts\activate        # Windows


pip install requests         # example package


# Step 5: Save your dependencies (every real project has this)
pip freeze > requirements.txt


# Step 6: Anyone can recreate your environment with:
pip install -r requirements.txt

# Step 7: Deactivate when done
deactivate
